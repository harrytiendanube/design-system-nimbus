## With prepend / append label

<Preview withToolbar>
  <Story name="with prepend / append label">
    {() => {
      const [values, setValues] = React.useState({ a: "", b: "", c: "" });
      const handleChange = (event) => {
        const { name, value } = event;
        setValues((prevState) => ({
          ...prevState,
          [name]: value,
        }));
      };
      return (
        <>
          <Input
            label="Label"
            prependLabel={select("prependLabel", ["$", "R$", "U$S"], "$")}
            appendLabel={select("appendLabel", ["cm", "kg"], "cm")}
            inputMode="decimal"
            value={values.a}
            placeholder="placeholder"
            name="a"
            error={text("error", "")}
            success={text("success", "")}
            isLoading={boolean("isLoading", true)}
            onChange={handleChange}
            disabled={boolean("disabled", false)}
          />
          <br />
          <Input
            label="Label"
            prependLabel={select("prependLabel2", ["$", "R$", "U$S"], "U$S")}
            inputMode="decimal"
            value={values.b}
            placeholder="placeholder"
            name="b"
            error={text("error", "")}
            success={text("success", "")}
            isLoading={boolean("isLoading", true)}
            onChange={handleChange}
            disabled={boolean("disabled", false)}
          />
          <br />
          <Input
            label="Label"
            appendLabel={select("appendLabel2", ["cm", "kg"], "kg")}
            inputMode="decimal"
            value={values.c}
            placeholder="placeholder"
            name="c"
            error={text("error", "")}
            success={text("success", "")}
            isLoading={boolean("isLoading", true)}
            onChange={handleChange}
            disabled={boolean("disabled", false)}
          />
        </>
      );
    }}
  </Story>
</Preview>
