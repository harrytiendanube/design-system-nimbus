  <Story name="default" parameters={{ decorators: [withKnobs, withA11y] }}>
    {() => {
      const [values, setValues] = React.useState({
        user: "",
        admin: "",
        email: "",
        pattern: "",
        price: "",
      });
      const handleChange = (event) => {
        const { name, value } = event;
        setValues((prevState) => ({
          ...prevState,
          [name]: value,
        }));
      };
      return (
        <Form
          textValidation={{
            email: "custom email validation message",
          }}
          alertText={text("alertText", "Alert message")}
          alertAppearance={select("alertAppearance", [
            "primary",
            "secondary",
            "danger",
            "success",
            "warning",
          ])}
          link={{
            children: "myLink",
            href: "https://www.tiendanube.com",
            onClick: action("onClick"),
          }}
          submitLabel="OK"
          onClickSubmit={() => {
            alert("click in Submit Button - Form is correctly validated");
          }}
          buttonLabel="Cancel"
          onClickButton={() => alert("click in Cancel Button")}
        >
          <InputValidator
            label="User"
            placeholder="Insert user"
            name="user"
            value={values.user}
            required
            onChange={handleChange}
          />
          <InputValidator
            label="Admin"
            placeholder="Insert Admin"
            name="admin"
            value={values.admin}
            maxLength={10}
            minLength={5}
            onChange={handleChange}
          />
          <InputValidator
            label="Email"
            placeholder="Insert Email"
            name="email"
            value={values.email}
            type="email"
            onChange={handleChange}
          />
          <InputValidator
            label="Pattern"
            placeholder="Type something that contains 1"
            name="pattern"
            value={values.pattern}
            type="text"
            pattern="1"
            onChange={handleChange}
          />
          <InputValidator
            prependLabel="$"
            label="Price"
            name="price"
            value={values.price}
            type="text"
            inputMode="decimal"
            pattern="^\d*(\.\d{1,3})?$"
            onChange={handleChange}
          />
        </Form>
      );
    }}
  </Story>
